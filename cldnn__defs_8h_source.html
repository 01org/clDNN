99  // There is no portable half precision floating point support.
100  // Using wrapped integral type with the same size and alignment restrictions.
101  class half_impl
102  {
103  public:
104  half_impl() = default;
105  template <typename T, typename = typename std::enable_if<!std::is_floating_point<T>::value>::type>
106  explicit half_impl(T data) : _data(data) {}
107 
108  operator uint16_t() const { return _data; }
109  operator float() const
110  {
111  cldnn_status status = CLDNN_SUCCESS;
112  auto value = cldnn_half_to_float(_data, &status);
113  if (status != CLDNN_SUCCESS)
114  throw std::runtime_error("Conversion from half failed");
115  return value;
116  }
117  explicit half_impl(float value)
118  {
119  cldnn_status status = CLDNN_SUCCESS;
120  _data = cldnn_float_to_half(value, &status);
121  if (status != CLDNN_SUCCESS)
122  throw std::runtime_error("Conversion to half failed");
123  }
124 
125  private:
126  uint16_t _data;
127  };
128 }

229 constexpr auto ceil_div(T1 val, T2 divider)
230  -> typename std::enable_if<std::is_integral<T1>::value && std::is_integral<T2>::value,
231  decltype(std::declval<typename std::make_unsigned<T1>::type>() / std::declval<typename std::make_unsigned<T2>::type>())>::type
