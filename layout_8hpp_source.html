107 constexpr auto fuse(data_types dt, cldnn::format::type fmt) -> decltype(static_cast<std::underlying_type<data_types>::type>(dt) | static_cast<std::underlying_type<format::type>::type>(fmt))
108 {
109  using dt_type = std::underlying_type<data_types>::type;
110  using fmt_type = std::underlying_type<cldnn::format::type>::type;
111  using fmt_narrow_type = int16_t;
112 
113  return static_cast<fmt_type>(fmt) <= std::numeric_limits<fmt_narrow_type>::max() &&
114  static_cast<dt_type>(dt) <= (std::numeric_limits<dt_type>::max() >> (sizeof(fmt_narrow_type) * 8))
115  ? (static_cast<dt_type>(dt) << (sizeof(fmt_narrow_type) * 8)) |
116  (static_cast<fmt_type>(fmt) >= 0 ? static_cast<fmt_narrow_type>(fmt) : static_cast<fmt_narrow_type>(-1))
117  : throw std::invalid_argument("data_type and/or format values are too big to be fused into single value");
118 }

286  size_t get_linear_size() const
287  {
288  auto sizes = get_buffer_size().sizes();
289  if (this->format == cldnn::format::os_iyx_osv16 && !is_aligned_to(sizes[0], 16))
290  {
291  sizes[0] = align_to(sizes[0], 16);
292  }
293  else if (this->format == cldnn::format::bs_xs_xsv8_bsv8 && !(is_aligned_to(sizes[0], 8) && is_aligned_to(sizes[2], 8)))
294  {
295  sizes[0] = align_to(sizes[0], 8);
296  sizes[2] = align_to(sizes[2], 8);
297  }
298  else if (this->format == cldnn::format::bs_xs_xsv8_bsv16 && !(is_aligned_to(sizes[0], 16) && is_aligned_to(sizes[2], 8)))
299  {
300  sizes[0] = align_to(sizes[0], 16);
301  sizes[2] = align_to(sizes[2], 8);
302  }
303  else if (this->format == cldnn::format::bs_x_bsv16 && !is_aligned_to(sizes[0], 16))
304  {
305  sizes[0] = align_to(sizes[0], 16);
306  }
307  return std::accumulate(
308  sizes.begin(),
309  sizes.end(),
310  static_cast<size_t>(1),
311  std::multiplies<size_t>()
312  );
