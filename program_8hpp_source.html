145  static std::vector<cldnn_primitive_id> to_refs(const std::vector<primitive_id>& stor)
146  {
147  std::vector<cldnn_primitive_id> result(stor.size());
148  for (size_t i = 0; i < stor.size(); i++)
149  {
150  result[i] = stor[i].c_str();

152  return std::move(result);
153  }
154 
155  static std::vector<primitive_id> make_outputs_from_ref(const cldnn_build_option& value)
156  {
157  if (value.type != cldnn_build_option_outputs) throw std::invalid_argument("option type does not match: should be 'output'");
158  if (value.data == nullptr) throw std::invalid_argument("output data is empty");
159  auto refs = reinterpret_cast<const cldnn_primitive_id_arr*>(value.data);
160  std::vector<primitive_id> result;
161  result.reserve(refs->size);
162  for (decltype(refs->size) i = 0; i < refs->size; i++)
163  {
164  result.push_back(refs->data[i]);
165  }
166  return result;
167  }

300  template<build_option_type OptType>
301  std::shared_ptr<const typename detail::build_option_traits<OptType>::object_type>
302  get()
303  {
304  using T = typename detail::build_option_traits<OptType>::object_type;
305  for (auto& option : _options)
306  {
307  if (option->get_type() == OptType)
308  return std::static_pointer_cast<const T>(option);
309  }
310  return std::static_pointer_cast<const T>(detail::build_option_traits<OptType>::make_default());
311  }

319  std::vector<cldnn_build_option> get_refs() const
320  {
321  std::vector<cldnn_build_option> result;
322  for (auto& o : _options)
323  {
324  result.push_back({ static_cast<int32_t>(o->get_type()), o->get_data() });
325  }
326  return result;
327  }
328 
329  void add_or_replace_option(std::shared_ptr<const build_option> opt)
330  {
331  for (auto& p : _options)
332  {
333  if (p->get_type() == opt->get_type())
334  {
335  p = opt;
336  return;
337  }
338  }
339  _options.push_back(opt);
340  }
341 
342  static std::shared_ptr<const build_option> make_option(const cldnn_build_option& option)
343  {
344  switch (option.type)
345  {
346  case cldnn_build_option_fusing:
347  return detail::build_option_traits<build_option_type::fusing>::make_option(option);
348  case cldnn_build_option_profiling:
349  return detail::build_option_traits<build_option_type::profiling>::make_option(option);
350  case cldnn_build_option_optimize_data:
351  return detail::build_option_traits<build_option_type::optimize_data>::make_option(option);
352  case cldnn_build_option_debug:
353  return detail::build_option_traits<build_option_type::debug>::make_option(option);
354  case cldnn_build_option_outputs:
355  return detail::build_option_traits<build_option_type::outputs>::make_option(option);
356  default: throw std::out_of_range("unsupported build option type");
357  }
358  }
