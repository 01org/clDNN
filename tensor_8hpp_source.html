499  tensor transform(cldnn::format new_fmt, value_type default_size) const
500  {
501  cldnn::format format = cldnn::format::bfyx;
502  auto val_order = format.internal_order();
503  auto new_order = new_fmt.internal_order();
504  std::vector<value_type> old_sizes = sizes();
505  std::vector<value_type> new_sizes(old_sizes.size(), default_size);
506  auto tmp = 1;
507  for(size_t i = 0; i < format.order().size(); i++)
508  {
509  auto c = val_order[i];
510  //skip f and y for the formats that do not have it
511  if (((new_fmt == format::bs_xs_xsv8_bsv8) || (new_fmt == format::bs_x_bsv16)) && ((c == 'f') || (c == 'y')))
512  {
513  if (new_order[i] == '?')
514  new_sizes[i] = default_size;
515 
516  tmp *= old_sizes[i];
517  continue;
518  }
519 
520  auto new_pos = new_order.find(c);
521  if (new_pos == std::string::npos)
522  throw std::invalid_argument("cannot convert to new format");
523  new_sizes[new_pos] = old_sizes[i];
524  }
525 
526  //in case of formats with smaller number of dimensions than input, flatten is performed below
527  if (tmp != 1)
528  {
529  for (size_t i = 0; i < format.order().size(); i++)
530  {
531  auto c = val_order[i];
532  if (c == 'x')
533  {
534  auto new_pos = new_order.find(c);
535  new_sizes[new_pos] *= tmp;
536  }
537  }
538  }
539  return { new_sizes };
540  }

544  size_t get_linear_offset(const tensor& coord, cldnn::format fmt) const
545  {
546  auto my_sizes = this->sizes();
547  auto adjusted_coords = coord.sizes();
548  if (fmt == cldnn::format::os_iyx_osv16 && !is_aligned_to(my_sizes[0], 16))
549  {
550  my_sizes[0] = align_to(my_sizes[0], 16);
551  adjusted_coords[0] = align_to(adjusted_coords[0], 16);
552  }
553  else if (fmt == cldnn::format::bs_xs_xsv8_bsv8 && !(is_aligned_to(my_sizes[0], 8) && is_aligned_to(my_sizes[1], 8)))
554  {
555  my_sizes[0] = align_to(my_sizes[0], 8);
556  my_sizes[1] = align_to(my_sizes[1], 8);
557  adjusted_coords[0] = align_to(adjusted_coords[0], 8);
558  adjusted_coords[1] = align_to(adjusted_coords[1], 8);
559  }
560  else if (fmt == cldnn::format::bs_x_bsv16 && !is_aligned_to(my_sizes[0], 16))
561  {
562  my_sizes[0] = align_to(my_sizes[0], 16);
563  adjusted_coords[0] = align_to(adjusted_coords[0], 16);
564  }
565 
566  assert(my_sizes.size() == adjusted_coords.size());
567 
568  assert(adjusted_coords.size() > 0);
569  size_t offset = adjusted_coords[0];
570  for(size_t i = 1; i < adjusted_coords.size(); i++ )
571  {
572  offset = offset * my_sizes[i - 1] + adjusted_coords[i];
573  }
574  return offset;
575  }
